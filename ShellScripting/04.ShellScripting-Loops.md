## **Part 1: For Loops**

### **Traditional For Loop**

```bash
#!/bin/bash

# C-style for loop
for (( i = 0; i < 5; i++ )); do
    echo "Iteration $i"
done

# Countdown
for (( i = 5; i > 0; i-- )); do
    echo "$i..."
done
echo "Blast off!"
```

### **For-In Loop (Iterating over List)**

```bash
#!/bin/bash

# Loop over list
for fruit in apple banana orange grape; do
    echo "I like $fruit"
done

# Loop over array
numbers=(1 2 3 4 5)
for num in ${numbers[@]}; do
    echo "Number: $num"
done

# Loop over range
for i in {1..5}; do
    echo "Count: $i"
done
```

### **For Loop with Step**

```bash
#!/bin/bash

# Loop with step
for i in {1..10..2}; do
    echo "$i"
done
# Output: 1 3 5 7 9

# Reverse with step
for i in {10..1..2}; do
    echo "$i"
done
# Output: 10 8 6 4 2
```

### **For Loop with Command Substitution**

```bash
#!/bin/bash

# Loop over command output
for file in $(ls -1 /tmp); do
    echo "File: $file"
done

# Loop over lines in file
for line in $(cat /etc/hostname); do
    echo "Hostname: $line"
done

# Loop over words
for word in $(echo "Hello World from Bash"); do
    echo "Word: $word"
done
```

## **Part 2: While Loops**

### **Basic While Loop**

```bash
#!/bin/bash

# Simple while loop
count=1
while [ $count -le 5 ]; do
    echo "Count: $count"
    ((count++))
done

# While with condition
i=0
while [ $i -lt 10 ]; do
    if [ $((i % 2)) -eq 0 ]; then
        echo "$i is even"
    fi
    ((i++))
done
```

### **While Reading Files**

```bash
#!/bin/bash

# Read file line by line
while read line; do
    echo "Line: $line"
done < /etc/hostname

# Read with IFS
while IFS=: read username password uid gid; do
    echo "User: $username (UID: $uid)"
done < /etc/passwd
```

### **While with User Input**

```bash
#!/bin/bash

# Read user input in loop
count=0
while true; do
    read -p "Enter something (type 'quit' to exit): " input
    
    if [ "$input" = "quit" ]; then
        echo "Goodbye!"
        break
    fi
    
    ((count++))
    echo "You entered: $input (Entry #$count)"
done
```

## **Part 3: Until Loops**

### **Basic Until Loop**

```bash
#!/bin/bash

# Until is opposite of while
count=1
until [ $count -gt 5 ]; do
    echo "Count: $count"
    ((count++))
done

# Wait for file to exist
until [ -f "/tmp/myfile.txt" ]; do
    echo "Waiting for file..."
    sleep 2
done
echo "File found!"
```

## **Part 4: Loop Control Statements**

### **Break Statement**

```bash
#!/bin/bash

# Break exits loop
for i in {1..10}; do
    if [ $i -eq 5 ]; then
        echo "Breaking at $i"
        break
    fi
    echo "Number: $i"
done
echo "Loop finished"

# Break from nested loop
for i in {1..3}; do
    for j in {1..3}; do
        if [ $j -eq 2 ]; then
            break 2  # Break both loops
        fi
        echo "i=$i j=$j"
    done
done
```

### **Continue Statement**

```bash
#!/bin/bash

# Continue skips to next iteration
for i in {1..5}; do
    if [ $i -eq 3 ]; then
        echo "Skipping $i"
        continue
    fi
    echo "Number: $i"
done

# Skip even numbers
for i in {1..10}; do
    if [ $((i % 2)) -eq 0 ]; then
        continue
    fi
    echo "Odd number: $i"
done
```

## **Part 5: Nested Loops**

### **Multiple Nested Loops**

```bash
#!/bin/bash

# Multiplication table
for i in {1..5}; do
    for j in {1..5}; do
        result=$((i * j))
        echo -n "$result "
    done
    echo
done

# Pattern with nested loops
for i in {1..3}; do
    for j in $(seq 1 $i); do
        echo -n "* "
    done
    echo
done
```

## **Part 6: Advanced Loop Patterns**

### **Loop with Array Index**

```bash
#!/bin/bash

declare -a colors=("red" "green" "blue" "yellow")

# Loop with index
for i in "${!colors[@]}"; do
    echo "$i: ${colors[$i]}"
done

# Loop with range
for ((i = 0; i < ${#colors[@]}; i++)); do
    echo "Color $((i+1)): ${colors[$i]}"
done
```

### **Loop with Multiple Conditions**

```bash
#!/bin/bash

# For loop with complex condition
for file in /tmp/*; do
    if [ -f "$file" ] && [ -r "$file" ]; then
        echo "Reading: $file"
        head -n 1 "$file"
    fi
done

# While with multiple operations
counter=1
while [ $counter -le 10 ] && [ $counter -gt 0 ]; do
    echo "Counter: $counter"
    if [ $counter -eq 5 ]; then
        echo "Halfway there!"
    fi
    ((counter++))
done
```

## **Part 7: Infinite Loops**

### **Creating Infinite Loops**

```bash
#!/bin/bash

# Infinite loop with break
while true; do
    read -p "Enter 'exit' to quit: " input
    if [ "$input" = "exit" ]; then
        break
    fi
    echo "You entered: $input"
done

# Infinite loop with for
for (( ; ; )); do
    echo "Press Ctrl+C to stop"
    sleep 1
done
```

## **Troubleshooting Guide**

| Issue | Solution |
|-------|----------|
| Infinite loop | Add break condition or Ctrl+C to stop |
| Loop not executing | Check loop condition syntax |
| Variable not updating | Use `(( ))` or `let` for arithmetic |
| Break/Continue not working | Ensure in correct loop context |
| Loop over files fails | Quote variables: `"$file"` |

---

## **Best Practices**

1. **Use meaningful loop variables** - `i` for index, `file` for filename
2. **Quote variables** in loops: `"$var"`
3. **Use appropriate loop type** - for lists, while for conditions
4. **Initialize variables** before loop
5. **Provide loop exit conditions** - avoid infinite loops
6. **Use descriptive comments** for complex loops
7. **Test loop logic** with small samples first
8. **Indent properly** for readability
9. **Use `((i++))` instead of `i=$((i+1))`** for clarity
10. **Handle empty lists** gracefully in for loops
