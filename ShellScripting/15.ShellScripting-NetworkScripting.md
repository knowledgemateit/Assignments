## **Part 2: Network Connectivity Checker**

### **Connectivity Testing Script**

```bash
#!/bin/bash

check_connectivity() {
    local host=$1
    local port=${2:-80}
    
    timeout 5 bash -c "cat < /dev/null > /dev/tcp/$host/$port" 2>/dev/null
    if [ $? -eq 0 ]; then
        return 0
    else
        return 1
    fi
}

monitor_hosts() {
    local hosts=("example.com" "google.com" "github.com")
    
    for host in "${hosts[@]}"; do
        if check_connectivity "$host"; then
            echo "✓ $host is reachable"
        else
            echo "✗ $host is unreachable"
        fi
    done
}

monitor_hosts
```

## **Part 3: Network Automation**

### **Port Status Checker**

```bash
#!/bin/bash

check_port() {
    local host=$1
    local port=$2
    
    if timeout 2 bash -c "cat < /dev/null > /dev/tcp/$host/$port" 2>/dev/null; then
        echo "Port $port on $host is OPEN"
        return 0
    else
        echo "Port $port on $host is CLOSED"
        return 1
    fi
}

monitor_services() {
    local services=(
        "example.com:80"
        "example.com:443"
        "db.example.com:3306"
    )
    
    for service in "${services[@]}"; do
        IFS=: read host port <<< "$service"
        check_port "$host" "$port"
    done
}
```

## **Part 4: Network Statistics**

### **Bandwidth Monitoring**

```bash
#!/bin/bash

monitor_bandwidth() {
    echo "Network Interface Statistics"
    echo "=============================="
    
    # Get interface statistics
    if command -v ifstat &>/dev/null; then
        ifstat -i eth0 1 3
    else
        # Alternative using ifconfig
        ifconfig | grep -E "RX|TX"
    fi
}

monitor_connections() {
    echo "Active Network Connections"
    echo "=========================="
    
    netstat -an | grep ESTABLISHED | wc -l
    echo "established connections"
}

monitor_bandwidth
monitor_connections
```
