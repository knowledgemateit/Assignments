## **Part 1: Basic Testing**

### **Testing Framework**

```bash
#!/bin/bash

# Simple test framework
run_test() {
    local name=$1
    local expected=$2
    local actual=$3
    
    if [ "$expected" = "$actual" ]; then
        echo "✓ PASS: $name"
        return 0
    else
        echo "✗ FAIL: $name"
        echo "  Expected: $expected"
        echo "  Actual: $actual"
        return 1
    fi
}

# Usage
result=$(add 2 3)
run_test "add 2+3" "5" "$result"
```

## **Part 2: Testing Functions**

### **Unit Tests**

```bash
#!/bin/bash

# Function to test
add_numbers() {
    local a=$1
    local b=$2
    echo $((a + b))
}

# Test cases
test_add() {
    local result
    
    result=$(add_numbers 2 3)
    [ "$result" -eq 5 ] && echo "✓ 2+3=5" || echo "✗ 2+3 failed"
    
    result=$(add_numbers 0 0)
    [ "$result" -eq 0 ] && echo "✓ 0+0=0" || echo "✗ 0+0 failed"
    
    result=$(add_numbers -5 5)
    [ "$result" -eq 0 ] && echo "✓ -5+5=0" || echo "✗ -5+5 failed"
}

test_add
```

## **Part 4: Integration Testing**

### **Testing Script Behavior**

```bash
#!/bin/bash

# Integration test
test_backup_script() {
    local test_dir=$(mktemp -d)
    local backup_dir=$(mktemp -d)
    
    trap "rm -rf $test_dir $backup_dir" EXIT
    
    # Create test files
    echo "test content" > "$test_dir/file1.txt"
    echo "test content" > "$test_dir/file2.txt"
    
    # Run backup
    ./backup.sh "$test_dir" "$backup_dir"
    
    # Verify
    [ -f "$backup_dir/backup.tar.gz" ] && echo "✓ Backup created"
    
    # Extract and verify
    tar -tzf "$backup_dir/backup.tar.gz" | grep -q "file1.txt" && echo "✓ Files backed up"
}

test_backup_script
```
