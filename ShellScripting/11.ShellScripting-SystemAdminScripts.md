```bash

INTERVAL=5

while true; do
    clear
    echo "=== Performance Monitor ==="
    echo "Time: $(date)"
    echo

    echo "CPU Usage:"
    ps aux | sort -nrk 3,3 | head -5
    echo
    
    # Memory
    echo "Memory Usage:"
    ps aux | sort -nrk 4,4 | head -5

    df -h | awk '$5+0 > 80 {print}'
    
    echo
    echo "Updating in $INTERVAL seconds (Ctrl+C to exit)..."
    sleep $INTERVAL
done
```

## **Part 2: User Management Scripts**

### **User Account Manager**

```bash
#!/bin/bash

# User Management Menu
show_menu() {
    echo "=== User Management ==="
    echo "1. Create user"
    echo "2. Delete user"
    echo "3. List users"
    echo "4. Change password"
    echo "5. Exit"
    echo -n "Choose option: "
}

create_user() {
    read -p "Username: " username
    if id "$username" &>/dev/null; then
        echo "User already exists"
        return 1
    fi
    sudo useradd -m -s /bin/bash "$username"
    echo "User $username created"
}

delete_user() {
    read -p "Username: " username
    if ! id "$username" &>/dev/null; then
        echo "User not found"
        return 1
    fi
    sudo userdel -r "$username"
    echo "User $username deleted"
}

list_users() {
    echo "=== System Users ==="
    cut -d: -f1,3,6 /etc/passwd | grep -E ":[0-9]{3}:" | column -t -s:
}

# Main loop
while true; do
    show_menu
    read -r choice
    case $choice in
        1) create_user ;;
        2) delete_user ;;
        3) list_users ;;
        4) read -p "Username: " user; sudo passwd "$user" ;;
        5) exit 0 ;;
        *) echo "Invalid option" ;;
    esac
done
```

## **Part 3: Backup and Recovery Scripts**

### **Simple Backup Script**

```bash
#!/bin/bash

# Directory Backup Script
set -euo pipefail

BACKUP_DIR="/backup"
SOURCE_DIR="${1:-.}"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="$BACKUP_DIR/backup_$TIMESTAMP.tar.gz"

# Validate
if [ ! -d "$SOURCE_DIR" ]; then
    echo "Error: Source directory not found: $SOURCE_DIR" >&2
    exit 1
fi

mkdir -p "$BACKUP_DIR"

# Create backup
echo "Creating backup of $SOURCE_DIR..."
tar -czf "$BACKUP_FILE" "$SOURCE_DIR"

if [ -f "$BACKUP_FILE" ]; then
    size=$(du -h "$BACKUP_FILE" | cut -f1)
    echo "Backup successful: $BACKUP_FILE ($size)"
else
    echo "Backup failed" >&2
    exit 1
fi
```

### **Incremental Backup**

```bash
#!/bin/bash

# Incremental Backup Script
set -euo pipefail

BACKUP_DIR="/backup/incremental"
SOURCE_DIR="${1:-.}"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)

mkdir -p "$BACKUP_DIR"

# Full backup on Monday
if [ "$(date +%A)" = "Monday" ]; then
    BACKUP_FILE="$BACKUP_DIR/full_$TIMESTAMP.tar.gz"
    tar -czf "$BACKUP_FILE" "$SOURCE_DIR"
    touch "$BACKUP_DIR/.lastfull"
    echo "Full backup: $BACKUP_FILE"
else
    # Incremental backup on other days
    LAST_FULL=$(stat -c %Y "$BACKUP_DIR/.lastfull" 2>/dev/null || echo 0)
    BACKUP_FILE="$BACKUP_DIR/incr_$TIMESTAMP.tar.gz"
    tar -czf "$BACKUP_FILE" --newer-mtime-than "$LAST_FULL" "$SOURCE_DIR"
    echo "Incremental backup: $BACKUP_FILE"
fi
```

### **Restore from Backup**

```bash
#!/bin/bash

# Restore from Backup Script
set -euo pipefail

BACKUP_FILE="${1:?Backup file required}"
RESTORE_DIR="${2:-.}"

if [ ! -f "$BACKUP_FILE" ]; then
    echo "Error: Backup file not found: $BACKUP_FILE" >&2
    exit 1
fi

echo "Restoring from $BACKUP_FILE to $RESTORE_DIR..."
mkdir -p "$RESTORE_DIR"
tar -xzf "$BACKUP_FILE" -C "$RESTORE_DIR"
echo "Restore complete"
```

## **Part 4: Log Analysis Scripts**

### **Log Monitor**

```bash
#!/bin/bash

# Log Monitor Script
LOGFILE="${1:-/var/log/syslog}"
ERROR_COUNT=0
WARNING_COUNT=0
INFO_COUNT=0

if [ ! -f "$LOGFILE" ]; then
    echo "Error: Log file not found: $LOGFILE" >&2
    exit 1
fi

echo "=== Log Analysis: $LOGFILE ==="
echo

# Count log levels
ERROR_COUNT=$(grep -ci "error" "$LOGFILE" || true)
WARNING_COUNT=$(grep -ci "warning" "$LOGFILE" || true)
INFO_COUNT=$(grep -ci "info" "$LOGFILE" || true)

echo "ERROR messages: $ERROR_COUNT"
echo "WARNING messages: $WARNING_COUNT"
echo "INFO messages: $INFO_COUNT"
echo

# Recent errors
echo "Recent errors:"
grep -i "error" "$LOGFILE" | tail -5

echo
echo "Recent warnings:"
grep -i "warning" "$LOGFILE" | tail -5
```

### **Log Rotation**

```bash
#!/bin/bash

# Log Rotation Script
set -euo pipefail

LOGDIR="/var/log/app"
MAX_SIZE=$((10 * 1024 * 1024))  # 10MB
KEEP_FILES=10

for logfile in "$LOGDIR"/*.log; do
    if [ ! -f "$logfile" ]; then
        continue
    fi
    
    size=$(stat -c%s "$logfile")
    if [ $size -gt $MAX_SIZE ]; then
        # Rotate log
        timestamp=$(date +%Y%m%d_%H%M%S)
        mv "$logfile" "${logfile}.${timestamp}"
        
        # Gzip old log
        gzip "${logfile}.${timestamp}"
        
        # Touch new log
        touch "$logfile"
    fi
    
    # Remove old logs
    ls -1 "${logfile}"* 2>/dev/null | sort -r | tail -n +$((KEEP_FILES+1)) | xargs -r rm
done

echo "Log rotation complete"
```

## **Part 5: Maintenance and Cleanup Scripts**

### **Cleanup Old Files**

```bash
#!/bin/bash

# Cleanup Old Files Script
set -euo pipefail

CLEANUP_DIR="${1:-.}"
DAYS="${2:-30}"
DRY_RUN="${3:-0}"

echo "Cleaning files older than $DAYS days in $CLEANUP_DIR"

if [ "$DRY_RUN" = "1" ]; then
    echo "DRY RUN MODE"
    find "$CLEANUP_DIR" -type f -mtime +$DAYS -exec ls -lh {} \;
    echo
    echo "Would remove the above files"
else
    count=$(find "$CLEANUP_DIR" -type f -mtime +$DAYS | wc -l)
    find "$CLEANUP_DIR" -type f -mtime +$DAYS -delete
    echo "Removed $count files"
fi
```

### **Disk Space Checker**

```bash
#!/bin/bash

# Disk Space Alert Script
THRESHOLD="${1:-80}"

echo "Checking disk usage (threshold: $THRESHOLD%)"
echo

df -h | awk -v threshold=$THRESHOLD '
    NR > 1 {
        usage = $5
        gsub(/%/, "", usage)
        if (usage+0 > threshold+0) {
            print "ALERT: " $6 " is " $5 " full (" $4 " available)"
        }
    }
'
```

### **System Update Script**

```bash
#!/bin/bash

# System Update Script
set -euo pipefail

echo "Updating system packages..."

if command -v apt &>/dev/null; then
    # Debian/Ubuntu
    sudo apt update
    sudo apt upgrade -y
    sudo apt autoremove -y

    # RedHat/CentOS
```

## **Best Practices**

1. **Test scripts thoroughly** before production use
2. **Use sudo carefully** - validate what requires elevation
3. **Back up before operations** - especially for destructive operations
4. **Log all operations** - for audit trail
5. **Use meaningful timestamps** - for traceability
6. **Validate all input** - even from trusted sources
7. **Handle errors gracefully** - don't continue on failure
8. **Document scripts** - explain what they do
9. **Version control scripts** - track changes
10. **Schedule with cron** - for automated tasks
