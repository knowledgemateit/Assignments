## **Project 1: Complete Infrastructure Management Suite**

### **Multi-Tool Infrastructure Manager**

```bash
#!/bin/bash

##############################################################
# Infrastructure Management Suite
# Complete system administration and monitoring
##############################################################

set -euo pipefail

# Core components
readonly CONFIG_DIR="${HOME}/.config/infra-mgmt"
readonly LOG_DIR="/var/log/infra-mgmt"
readonly BACKUP_DIR="/backup/infra-mgmt"

# Create directories
mkdir -p "$CONFIG_DIR" "$LOG_DIR" "$BACKUP_DIR"

# Logging system
log() {
    echo "[$(date +'%Y-%m-%d %H:%M:%S')] $*" | tee -a "$LOG_DIR/main.log"
}

error() {
    echo "[ERROR] $*" >&2 | tee -a "$LOG_DIR/error.log"
    exit 1
}

# System monitoring
monitor_system() {
    log "Monitoring system health..."
    
    # CPU check
    local cpu=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
    log "CPU Usage: ${cpu}%"
    
    # Memory check
    local mem=$(free | grep Mem | awk '{printf("%.0f", ($3/$2) * 100)}')
    log "Memory Usage: ${mem}%"
    
    # Disk check
    df -h | tail -n +2 | while read line; do
        log "Disk: $line"
    done
}

# Service management
manage_services() {
    local action=$1
    local services=("ssh" "cron" "syslog")
    
    for service in "${services[@]}"; do
        case "$action" in
            start|stop|restart)
                systemctl "$action" "$service"
                log "Service $service: $action"
                ;;
        esac
    done
}

# Backup management
backup_system() {
    log "Creating system backup..."
    local backup_file="$BACKUP_DIR/system_$(date +%Y%m%d_%H%M%S).tar.gz"
    tar -czf "$backup_file" /etc /home /var/www 2>/dev/null || true
    log "Backup created: $backup_file"
}

# Main menu
show_menu() {
    echo "=== Infrastructure Management Suite ==="
    echo "1. Monitor system"
    echo "2. Manage services"
    echo "3. Create backup"
    echo "4. View logs"
    echo "5. Exit"
}

# Main loop
main() {
    while true; do
        show_menu
        read -p "Choose option: " choice
        case $choice in
            1) monitor_system ;;
            2) manage_services "restart" ;;
            3) backup_system ;;
            4) tail -50 "$LOG_DIR/main.log" ;;
            5) exit 0 ;;
            *) echo "Invalid option" ;;
        esac
    done
}

main
```

## **Project 3: Advanced Monitoring and Alerting System**

### **Enterprise Monitoring Solution**

```bash
#!/bin/bash

# Advanced monitoring with alerting
declare -A thresholds=(
    [cpu]=80
    [mem]=85
    [disk]=90
)

check_thresholds() {
    local cpu=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
    
    if (( $(echo "$cpu > ${thresholds[cpu]}" | bc -l) )); then
        alert "CRITICAL" "CPU Usage: ${cpu}%"
    fi
}

alert() {
    local level=$1
    local message=$2
    
    echo "[$level] $message" | mail -s "Alert: $level" admin@example.com
}
```
