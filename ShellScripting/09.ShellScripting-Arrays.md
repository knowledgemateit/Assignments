## **Part 1: Indexed Arrays**

### **Creating Indexed Arrays**

```bash
#!/bin/bash

# Method 1: Declare and assign
declare -a fruits
fruits[0]="apple"
fruits[1]="banana"
fruits[2]="orange"

# Method 2: Initialize on declaration
fruits=("apple" "banana" "orange" "grape")

# Method 3: Initialize with values
declare -a colors=("red" "green" "blue")

# Access elements
echo "First fruit: ${fruits[0]}"
echo "Second fruit: ${fruits[1]}"
```

### **Array Operations**

```bash
#!/bin/bash

arr=(1 2 3 4 5)

# Get array length
echo "Length: ${#arr[@]}"

# Get specific element
echo "Element 2: ${arr[2]}"

# Get all elements
echo "All: ${arr[@]}"

# Get range
echo "Range 1-3: ${arr[@]:1:3}"

# Append element
arr+=(6)

# Update element
arr[2]=30

# Remove element by setting to null
unset arr[3]
```

### **Array Iteration**

```bash
#!/bin/bash

arr=("apple" "banana" "orange")

# Loop with for-in
for fruit in "${arr[@]}"; do
    echo "Fruit: $fruit"
done

# Loop with index
for i in "${!arr[@]}"; do
    echo "$i: ${arr[$i]}"
done

# Loop with C-style
for ((i = 0; i < ${#arr[@]}; i++)); do
    echo "${arr[$i]}"
done

# Loop with while
i=0
while [ $i -lt ${#arr[@]} ]; do
    echo "${arr[$i]}"
    ((i++))
done
```

## **Part 2: Associative Arrays**

### **Creating Associative Arrays**

```bash
#!/bin/bash

# Declare associative array
declare -A student

# Assign values with keys
student[name]="Khajanaraju"
student[age]="30"
student[city]="Bangalore"
student[email]="khajanaraju@example.com"

# Initialize on declaration
declare -A config=(
    [host]="localhost"
    [port]="8080"
    [debug]="true"
)

# Access elements
echo "Name: ${student[name]}"
echo "Age: ${student[age]}"
```

### **Associative Array Operations**

```bash
#!/bin/bash

declare -A person

# Assign values
person[first]="John"
person[last]="Doe"
person[age]="25"

# Get all values
echo "Values: ${person[@]}"

# Get all keys
echo "Keys: ${!person[@]}"

# Check if key exists
if [[ -v person[age] ]]; then
    echo "Age exists"
fi

# Delete element
unset person[age]

# Array length
echo "Size: ${#person[@]}"
```

### **Iterating Associative Arrays**

```bash
#!/bin/bash

declare -A config=(
    [host]="localhost"
    [port]="8080"
    [user]="admin"
    [debug]="true"
)

# Iterate over keys and values
for key in "${!config[@]}"; do
    echo "$key: ${config[$key]}"
done

# Iterate over values only
for value in "${config[@]}"; do
    echo "Value: $value"
done

# Sort keys
for key in $(printf '%s\n' "${!config[@]}" | sort); do
    echo "$key: ${config[$key]}"
done
```

## **Part 3: Array Functions**

### **Working with Arrays in Functions**

```bash
#!/bin/bash

# Pass array to function
function print_array() {
    local arr=("$@")
    echo "Array contents:"
    for item in "${arr[@]}"; do
        echo "  - $item"
    done
}

# Return array from function
function create_array() {
    local arr=(1 2 3 4 5)
    echo "${arr[@]}"
}

# Call functions
print_array "apple" "banana" "orange"

# Capture returned array
result=($(create_array))
echo "Result: ${result[@]}"
```

### **Array Processing Functions**

```bash
#!/bin/bash

# Sum array elements
function array_sum() {
    local sum=0
    for item in "$@"; do
        sum=$((sum + item))
    done
    echo $sum
}

# Find maximum
function array_max() {
    local max=$1
    for item in "$@"; do
        if [ $item -gt $max ]; then
            max=$item
        fi
    done
    echo $max
}

# Find minimum
function array_min() {
    local min=$1
    for item in "$@"; do
        if [ $item -lt $min ]; then
            min=$item
        fi
    done
    echo $min
}

# Test functions
sum=$(array_sum 1 2 3 4 5)
echo "Sum: $sum"

max=$(array_max 10 45 23 8 34)
echo "Max: $max"

min=$(array_min 10 45 23 8 34)
echo "Min: $min"
```

## **Part 4: Advanced Array Operations**

### **Search and Filter**

```bash
#!/bin/bash

# Search in array
function array_contains() {
    local search=$1
    shift
    for item in "$@"; do
        if [ "$item" = "$search" ]; then
            return 0
        fi
    done
    return 1
}

# Filter array
function array_filter() {
    local pattern=$1
    shift
    for item in "$@"; do
        if [[ "$item" =~ $pattern ]]; then
            echo "$item"
        fi
    done
}

# Join array
function array_join() {
    local delimiter=$1
    shift
    printf "%s" "$1"
    shift
    for item in "$@"; do
        printf "%s" "$delimiter$item"
    done
    echo
}

# Test operations
array_contains "banana" "apple" "banana" "orange" && echo "Found banana"
array_join "," apple banana orange
array_filter "^a" apple banana apricot orange
```

### **Array Manipulation**

```bash
#!/bin/bash

# Remove duplicates
function array_unique() {
    printf '%s\n' "$@" | sort -u
}

# Reverse array
function array_reverse() {
    local -a arr=("$@")
    for ((i = ${#arr[@]} - 1; i >= 0; i--)); do
        echo "${arr[$i]}"
    done
}

# Slice array
function array_slice() {
    local start=$1
    local length=$2
    shift 2
    for ((i = start; i < start + length && i < $#; i++)); do
        echo "${@:i+1:1}"
    done
}

# Test functions
array_unique apple banana apple orange
```

## **Troubleshooting Guide**

| Issue | Solution |
|-------|----------|
| Array not declared | Use `declare -a` for indexed or `-A` for associative |
| Index out of bounds | Check array length with `${#arr[@]}` |
| Associative array error | Bash 4+, not available in sh |
| Quote issues | Always quote: `"${arr[@]}"` |
| Unset element | Use `unset arr[index]` to remove |

---

## **Best Practices**

1. **Quote arrays properly** - `"${arr[@]}"` to preserve elements
2. **Use meaningful names** for array indices in associative arrays
3. **Declare arrays explicitly** - avoid implicit creation
4. **Check array length** before accessing by index
5. **Use appropriate array type** - indexed for lists, associative for records
6. **Initialize arrays** before use
7. **Pass arrays carefully** to functions with `"$@"`
8. **Validate input** before adding to arrays
9. **Clean up large arrays** when no longer needed
10. **Document array structure** in comments
