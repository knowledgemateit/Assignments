## **Part 1: Function Basics**

### **Defining Functions**

```bash
#!/bin/bash

# Method 1: Function keyword
function greet() {
    echo "Hello, World!"
}

# Method 2: Without function keyword
greet2() {
    echo "Hello again!"
}

# Call functions
greet
greet2
```

### **Function with Simple Logic**

```bash
#!/bin/bash

function calculate_sum() {
    local num1=10
    local num2=20
    local sum=$((num1 + num2))
    echo "Sum: $sum"
}

calculate_sum
```

## **Part 2: Function Parameters**

### **Positional Parameters**

```bash
#!/bin/bash

# Function with parameters
function add_numbers() {
    local a=$1
    local b=$2
    local sum=$((a + b))
    echo "Sum of $a and $b is: $sum"
}

# Call with arguments
add_numbers 10 20
add_numbers 5 15

# Function with multiple parameters
function introduce() {
    local name=$1
    local age=$2
    local city=$3
    echo "Hello, I'm $name, $age years old, from $city"
}

introduce "Khajanaraju" "30" "Bangalore"
```

### **Parameter Validation**

```bash
#!/bin/bash

function process_file() {
    if [ $# -lt 1 ]; then
        echo "Error: Filename required"
        return 1
    fi
    
    local file=$1
    if [ ! -f "$file" ]; then
        echo "Error: File not found: $file"
        return 2
    fi
    
    echo "Processing: $file"
    wc -l "$file"
}

process_file "/etc/hostname"
```

### **Using Special Variables in Functions**

```bash
#!/bin/bash

function display_args() {
    echo "Function name: $0"
    echo "First argument: $1"
    echo "All arguments: $@"
    echo "Number of arguments: $#"
    
    # Shift to remove first argument
    shift
    echo "After shift: $@"
}

display_args arg1 arg2 arg3
```

## **Part 3: Function Return Values**

### **Return Codes**

```bash
#!/bin/bash

function check_file() {
    local file=$1
    
    if [ -f "$file" ]; then
        return 0  # Success
    else
        return 1  # Failure
    fi
}

# Check return value
if check_file "/etc/hostname"; then
    echo "File exists"
else
    echo "File not found"
fi

# Check specific return code
check_file "/nonexistent"
if [ $? -eq 1 ]; then
    echo "File not found (error code 1)"
fi
```

### **Returning Values via Output**

```bash
#!/bin/bash

function get_square() {
    local num=$1
    local result=$((num * num))
    echo $result
}

# Capture function output
square=$(get_square 5)
echo "Square of 5: $square"

# Use in expressions
echo "Square of 10: $(get_square 10)"
```

### **Returning Multiple Values**

```bash
#!/bin/bash

function get_file_info() {
    local file=$1
    local size=$(wc -c < "$file")
    local lines=$(wc -l < "$file")
    
    # Return as space-separated values
    echo "$size $lines"
}

# Parse return values
info=$(get_file_info "/etc/hostname")
set -- $info
size=$1
lines=$2
echo "Size: $size, Lines: $lines"
```

## **Part 4: Function Scope**

### **Local vs Global Variables**

```bash
#!/bin/bash

# Global variable
global_var="I'm global"

function scope_demo() {
    # Local variable
    local local_var="I'm local"
    
    # Can access global
    echo "Global in function: $global_var"
    echo "Local: $local_var"
    
    # Modify global
    global_var="Modified globally"
}

echo "Before: $global_var"
scope_demo
echo "After: $global_var"
echo "Local outside: $local_var"  # Empty - not accessible
```

### **Function-Local Arrays**

```bash
#!/bin/bash

function array_scope() {
    local -a local_array=("a" "b" "c")
    echo "In function: ${local_array[@]}"
}

function array_global() {
    global_array=("x" "y" "z")
}

array_scope
array_global
echo "Global array: ${global_array[@]}"
```

## **Part 5: Advanced Function Patterns**

### **Recursive Functions**

```bash
#!/bin/bash

# Factorial using recursion
function factorial() {
    local n=$1
    
    if [ $n -le 1 ]; then
        echo 1
    else
        local prev=$(factorial $((n - 1)))
        echo $((n * prev))
    fi
}

echo "Factorial of 5: $(factorial 5)"
```

### **Functions Calling Other Functions**

```bash
#!/bin/bash

function validate_number() {
    local num=$1
    
    if ! [[ "$num" =~ ^[0-9]+$ ]]; then
        return 1
    fi
    return 0
}

function double_number() {
    if validate_number "$1"; then
        echo $((2 * $1))
    else
        echo "Error: Invalid number"
        return 1
    fi
}

double_number 5   # OK
double_number "abc"  # Error
```

### **Default Parameter Values**

```bash
#!/bin/bash

function greet_user() {

    local greeting=${2:-"Hello"}
    
    echo "$greeting, $name!"
}

greet_user

greet_user "Khajanaraju" "Welcome"
```

## **Part 6: Function Libraries**

### **Creating Reusable Function Library**

```bash
# File: lib/utils.sh
#!/bin/bash

# String functions
function uppercase() {
    echo "$1" | tr '[:lower:]' '[:upper:]'
}

function lowercase() {
    echo "$1" | tr '[:upper:]' '[:lower:]'
}

# File functions
function file_size() {
    if [ -f "$1" ]; then
        wc -c < "$1"
    fi
}

# Math functions
function add() {
    echo $(($1 + $2))
}

function subtract() {
    echo $(($1 - $2))
}
```

### **Using Function Library**

```bash
#!/bin/bash

# Source the library
source ./lib/utils.sh

# Use library functions
echo "Uppercase: $(uppercase 'hello')"
echo "Sum: $(add 10 20)"
echo "Difference: $(subtract 20 5)"
```

## **Part 7: Error Handling in Functions**

### **Function Error Handling**

```bash
#!/bin/bash

function safe_operation() {
    local file=$1
    
    if [ -z "$file" ]; then
        echo "Error: No file specified" >&2
        return 1
    fi
    
    if [ ! -f "$file" ]; then
        echo "Error: File not found: $file" >&2
        return 2
    fi
    
    cat "$file"
    return 0
}

# Use function with error handling
if safe_operation "/etc/hostname"; then
    echo "Operation successful"
else
    echo "Operation failed with code: $?"
fi
```

## **Troubleshooting Guide**

| Issue | Solution |
|-------|----------|
| Function not found | Ensure function defined before use |
| Parameters not passed | Use `$1`, `$2` not `$parameter` |
| Return value not captured | Use output redirection: `result=$(func)` |
| Local variables failing | Use `local` keyword in function |
| Recursive infinite loop | Add termination condition |

## **Best Practices**

1. **Use function names** that describe purpose
2. **Document functions** with comments
3. **Validate parameters** before use
4. **Use local variables** in functions
5. **Return meaningful exit codes** (0 for success)
6. **Send errors to stderr** with `>&2`
7. **Use function libraries** for code reuse
8. **Keep functions focused** - do one thing well
9. **Test functions** with various inputs
10. **Use meaningful parameter names** via comments
