## **Project 1: Deployment Automation Script**

### **Complete Deployment Manager**

```bash
#!/bin/bash

##############################################################
# Deployment Manager
# Purpose: Automate application deployment with rollback
# Usage: ./deploy.sh [env] [version] [--dry-run]
##############################################################

set -euo pipefail
IFS=$'\n\t'

# Configuration
readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly APP_DIR="/opt/app"
readonly BACKUP_DIR="/opt/backups"
readonly LOG_FILE="/var/log/deploy.log"
readonly TIMESTAMP=$(date +%Y%m%d_%H%M%S)

# Variables
ENVIRONMENT="${1:-staging}"
VERSION="${2:?Version required}"
DRY_RUN="${3:-}"

# Logging
log() {
    echo "[$(date +'%Y-%m-%d %H:%M:%S')] $*" | tee -a "$LOG_FILE"
}

error() {
    echo "[ERROR] $*" >&2 | tee -a "$LOG_FILE"
    exit 1
}

# Validation
validate_environment() {
    case "$ENVIRONMENT" in
        staging|production)
            return 0
            ;;
        *)
            error "Invalid environment: $ENVIRONMENT"
            ;;
    esac
}

validate_version() {
    if [ ! -f "$BACKUP_DIR/app-$VERSION.tar.gz" ]; then
        error "Version not found: $VERSION"
    fi
}

# Deployment
backup_current() {
    log "Creating backup of current version..."
    tar -czf "$BACKUP_DIR/current_$TIMESTAMP.tar.gz" "$APP_DIR"
    log "Backup created successfully"
}

deploy_version() {
    log "Deploying version $VERSION to $ENVIRONMENT..."
    
    if [ "$DRY_RUN" = "--dry-run" ]; then
        log "DRY RUN: Would deploy $VERSION"
        return 0
    fi
    
    tar -xzf "$BACKUP_DIR/app-$VERSION.tar.gz" -C /opt/
    log "Version $VERSION deployed"
}

verify_deployment() {
    log "Verifying deployment..."
    
    # Check application is running
    if pgrep -f "app-server" > /dev/null; then
        log "Application is running"
        return 0
    else
        error "Application failed to start"
    fi
}

rollback() {
    log "Rolling back to previous version..."
    tar -xzf "$BACKUP_DIR/current_$TIMESTAMP.tar.gz" -C /
    log "Rollback completed"
}

# Main
main() {
    log "=== Deployment Started ==="
    log "Environment: $ENVIRONMENT"
    log "Version: $VERSION"
    
    validate_environment
    validate_version
    backup_current
    deploy_version || { rollback; error "Deployment failed"; }
    verify_deployment
    
    log "=== Deployment Completed Successfully ==="
}

# Run
main "$@"
```

## **Project 2: Log Analysis and Reporting**

### **Smart Log Analyzer**

```bash
#!/bin/bash

##############################################################
# Log Analysis and Reporting
# Purpose: Analyze logs and generate reports
# Usage: ./log_analyzer.sh [logfile] [report_format]
##############################################################

set -euo pipefail

LOGFILE="${1:?Logfile required}"
FORMAT="${2:-text}"  # text, csv, json

# Validate input
if [ ! -f "$LOGFILE" ]; then
    echo "Error: Log file not found: $LOGFILE" >&2
    exit 1
fi

# Create report directory
REPORT_DIR="./reports"
mkdir -p "$REPORT_DIR"
REPORT_FILE="$REPORT_DIR/report_$(date +%Y%m%d_%H%M%S).$FORMAT"

# Analysis functions
analyze_errors() {
    grep -i "error\|failed\|exception" "$LOGFILE" | wc -l
}

analyze_warnings() {
    grep -i "warning\|warn" "$LOGFILE" | wc -l
}

analyze_info() {
    grep -i "info\|information" "$LOGFILE" | wc -l
}

get_error_types() {
    grep -i "error\|failed" "$LOGFILE" | \
        sed -E 's/.*\[(.*?)\].*/\1/' | \
        sort | uniq -c | sort -rn
}

get_top_errors() {
    grep -i "error\|failed" "$LOGFILE" | \
        sed -E 's/.*: (.*)/\1/' | \
        sort | uniq -c | sort -rn | head -10
}

# Report generation
generate_text_report() {
    {
        echo "=== LOG ANALYSIS REPORT ==="
        echo "Generated: $(date)"
        echo "Log File: $LOGFILE"
        echo
        
        echo "=== SUMMARY ==="
        echo "Total Lines: $(wc -l < "$LOGFILE")"
        echo "Error Count: $(analyze_errors)"
        echo "Warning Count: $(analyze_warnings)"
        echo "Info Count: $(analyze_info)"
        echo
        
        echo "=== TOP ERRORS ==="
        get_top_errors
        echo
        
        echo "=== ERROR TYPES ==="
        get_error_types
    } > "$REPORT_FILE"
}

generate_csv_report() {
    {
        echo "Metric,Count"
        echo "Total Lines,$(wc -l < "$LOGFILE")"
        echo "Errors,$(analyze_errors)"
        echo "Warnings,$(analyze_warnings)"
        echo "Info,$(analyze_info)"
    } > "$REPORT_FILE"
}

generate_json_report() {
    {
        echo "{"
        echo "  \"timestamp\": \"$(date -u +%Y-%m-%dT%H:%M:%SZ)\","
        echo "  \"logfile\": \"$LOGFILE\","
        echo "  \"summary\": {"
        echo "    \"totalLines\": $(wc -l < "$LOGFILE"),"
        echo "    \"errors\": $(analyze_errors),"
        echo "    \"warnings\": $(analyze_warnings),"
        echo "    \"info\": $(analyze_info)"
        echo "  }"
        echo "}"
    } > "$REPORT_FILE"
}

# Main
case "$FORMAT" in
    text)
        generate_text_report
        ;;
    csv)
        generate_csv_report
        ;;
    json)
        generate_json_report
        ;;
    *)
        echo "Unknown format: $FORMAT" >&2
        exit 1
        ;;
esac

echo "Report generated: $REPORT_FILE"
cat "$REPORT_FILE"
```

## **Project 3: System Health Dashboard**

### **Interactive Health Monitor**

```bash
#!/bin/bash

##############################################################
# System Health Dashboard
# Purpose: Real-time system monitoring
# Usage: ./health_dashboard.sh
##############################################################

set -euo pipefail

# Colors
RED='\033[0;31m'
YELLOW='\033[1;33m'
GREEN='\033[0;32m'
NC='\033[0m'  # No Color

# Thresholds
CPU_THRESHOLD=80
MEM_THRESHOLD=80
DISK_THRESHOLD=85

# Alert function
alert() {
    local level=$1
    local message=$2
    case "$level" in
        critical)
            echo -e "${RED}[CRITICAL]${NC} $message"
            ;;
        warning)
            echo -e "${YELLOW}[WARNING]${NC} $message"
            ;;
        ok)
            echo -e "${GREEN}[OK]${NC} $message"
            ;;
    esac
}

# Check CPU
check_cpu() {
    local cpu_usage=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
    if (( $(echo "$cpu_usage > $CPU_THRESHOLD" | bc -l) )); then
        alert critical "CPU usage: ${cpu_usage}%"
    else
        alert ok "CPU usage: ${cpu_usage}%"
    fi
}

# Check Memory
check_memory() {
    local mem_usage=$(free | grep Mem | awk '{printf("%.0f", ($3/$2) * 100)}')
    if [ "$mem_usage" -gt "$MEM_THRESHOLD" ]; then
        alert critical "Memory usage: ${mem_usage}%"
    else
        alert ok "Memory usage: ${mem_usage}%"
    fi
}

# Check Disk
check_disk() {
    while read -r line; do
        local usage=$(echo "$line" | awk '{print $5}' | cut -d'%' -f1)
        local mount=$(echo "$line" | awk '{print $6}')
        if [ "$usage" -gt "$DISK_THRESHOLD" ]; then
            alert critical "Disk usage on $mount: ${usage}%"
        fi
    done < <(df -h | tail -n +2)
}

# Check Services
check_services() {
    local services=("ssh" "cron" "syslog")
    for service in "${services[@]}"; do
        if systemctl is-active --quiet "$service"; then
            alert ok "$service is running"
        else
            alert warning "$service is not running"
        fi
    done
}

# Main dashboard
main() {
    while true; do
        clear
        echo "=== SYSTEM HEALTH DASHBOARD ==="
        echo "Time: $(date)"
        echo "Hostname: $(hostname)"
        echo "Uptime: $(uptime -p)"
        echo
        
        echo "=== SYSTEM STATUS ==="
        check_cpu
        echo
        check_memory
        echo
        echo "=== DISK STATUS ==="
        check_disk
        echo
        echo "=== SERVICE STATUS ==="
        check_services
        echo
        echo "Refreshing in 10 seconds... (Ctrl+C to exit)"
        sleep 10
    done
}

main
```

## **Project 4: Database Backup Manager**

### **Intelligent Backup System**

```bash
#!/bin/bash

##############################################################
# Database Backup Manager
# Purpose: Manage database backups with retention
# Usage: ./db_backup.sh [database] [backup_type]
##############################################################

set -euo pipefail

BACKUP_DIR="/var/backups/databases"
RETENTION_DAYS=30
LOG_FILE="/var/log/db_backup.log"

# Create backup directory
mkdir -p "$BACKUP_DIR"

log() {
    echo "[$(date +'%Y-%m-%d %H:%M:%S')] $*" | tee -a "$LOG_FILE"
}

error() {
    log "ERROR: $*"
    exit 1
}

# Backup functions
backup_mysql() {
    local db=$1
    local backup_file="$BACKUP_DIR/mysql_${db}_$(date +%Y%m%d_%H%M%S).sql.gz"
    
    log "Backing up MySQL database: $db"
    mysqldump "$db" | gzip > "$backup_file"
    log "Backup completed: $backup_file"
    echo "$backup_file"
}

backup_postgresql() {
    local db=$1
    local backup_file="$BACKUP_DIR/postgresql_${db}_$(date +%Y%m%d_%H%M%S).sql.gz"
    
    log "Backing up PostgreSQL database: $db"
    pg_dump "$db" | gzip > "$backup_file"
    log "Backup completed: $backup_file"
    echo "$backup_file"
}

# Cleanup old backups
cleanup_old_backups() {
    log "Cleaning up backups older than $RETENTION_DAYS days"
    find "$BACKUP_DIR" -name "*.sql.gz" -mtime +$RETENTION_DAYS -delete
    log "Cleanup completed"
}

# Verify backup
verify_backup() {
    local file=$1
    log "Verifying backup: $file"
    if gzip -t "$file" 2>/dev/null; then
        log "Backup verification passed"
        return 0
    else
        error "Backup verification failed: $file"
    fi
}

# Main
main() {
    local database="${1:?Database name required}"
    local type="${2:-mysql}"
    
    local backup_file
    case "$type" in
        mysql)
            backup_file=$(backup_mysql "$database")
            ;;
        postgresql|postgres)
            backup_file=$(backup_postgresql "$database")
            ;;
        *)
            error "Unknown database type: $type"
            ;;
    esac
    
    verify_backup "$backup_file"
    cleanup_old_backups
    
    log "Backup process completed"
}

main "$@"
```
