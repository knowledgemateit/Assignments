## **Part 1: Reading Files**

### **Reading Entire File**

```bash
#!/bin/bash

# Method 1: cat command
content=$(cat /etc/hostname)
echo "$content"

# Method 2: read builtin
while read line; do
    echo "Line: $line"
done < /etc/hostname

# Method 3: readfile function
readfile() {
    while read -r line; do
        echo "$line"
    done < "$1"
}

readfile /etc/hostname
```

### **Reading Specific Lines**

```bash
#!/bin/bash

file="/etc/hostname"

# First line
head -1 "$file"

# Last line
tail -1 "$file"

# Lines 5-10
sed -n '5,10p' "$file"

# Nth line
sed -n '3p' "$file"
```

### **Reading with Field Separator**

```bash
#!/bin/bash

# Read CSV file with comma separator
while IFS=, read field1 field2 field3; do
    echo "Field1: $field1, Field2: $field2"
done < data.csv

# Read /etc/passwd with colon separator
while IFS=: read username password uid gid; do
    echo "User: $username (UID: $uid)"
done < /etc/passwd
```

## **Part 2: Writing to Files**

### **Creating and Writing Files**

```bash
#!/bin/bash

# Method 1: Using echo with redirection (overwrites)
echo "Hello, World!" > output.txt

# Method 2: Using printf
printf "Line 1\nLine 2\n" > output.txt

# Method 3: Using cat with here-document
cat > output.txt << 'EOF'
This is line 1
This is line 2
This is line 3
EOF

# Method 4: Using tee
echo "Hello" | tee output.txt

# Store multiple values
{
    echo "Line 1"
    echo "Line 2"
    echo "Line 3"
} > output.txt
```

### **Appending to Files**

```bash
#!/bin/bash

# Append with >>
echo "New line" >> output.txt

# Append multiple lines
cat >> output.txt << 'EOF'
Line 1
Line 2
EOF

# Append from variable
message="Appended message"
echo "$message" >> output.txt
```

## **Part 3: File Manipulation**

### **Modifying File Content**

```bash
#!/bin/bash

file="test.txt"

# Create test file
echo -e "apple\nbanana\ncherry" > "$file"

# Replace text (uses sed)
sed 's/apple/apricot/' "$file" > temp.txt
mv temp.txt "$file"

# Delete lines containing pattern
grep -v "banana" "$file" > temp.txt
mv temp.txt "$file"

# Add line numbers
nl "$file"

# Count lines
wc -l "$file"
```

### **Working with File Names**

```bash
#!/bin/bash

file="/home/user/documents/report.txt"

# Get directory
dirname "$file"      # /home/user/documents

# Get filename
basename "$file"     # report.txt

# Get filename without extension
basename "$file" .txt  # report

# Get file extension
echo "${file##*.}"   # txt

# Get full path
realpath "$file"
```

## **Part 4: Directory Operations**

### **Creating and Managing Directories**

```bash
#!/bin/bash

# Create directory
mkdir -p /tmp/my_scripts

# Create nested directories
mkdir -p /tmp/dir1/dir2/dir3

# List directory contents
ls -la /tmp

# Count files in directory
ls -1 /tmp | wc -l

# List files recursively
find /tmp -type f

# Create files in directories
for dir in {1..3}; do
    mkdir -p "project$dir"
    touch "project$dir/file.txt"
done
```

### **Directory Traversal**

```bash
#!/bin/bash

# Process all files in directory
for file in /tmp/*; do
    if [ -f "$file" ]; then
        echo "File: $file"
        echo "Size: $(wc -c < "$file")"
    elif [ -d "$file" ]; then
        echo "Directory: $file"
    fi
done

# Recursive directory processing
process_dir() {
    local dir=$1
    for item in "$dir"/*; do
        if [ -f "$item" ]; then
            echo "Processing file: $item"
        elif [ -d "$item" ]; then
            process_dir "$item"
        fi
    done
}

process_dir /tmp
```

## **Part 5: File Properties and Testing**

### **Testing Files and Directories**

```bash
#!/bin/bash

file="/etc/hostname"

# File existence
if [ -e "$file" ]; then
    echo "File exists"
fi

# Is regular file
if [ -f "$file" ]; then
    echo "Is regular file"
fi

# Is directory
if [ -d "/etc" ]; then
    echo "Is directory"
fi

# File is readable
if [ -r "$file" ]; then
    echo "File is readable"
fi

# File is writable
if [ -w "$file" ]; then
    echo "File is writable"
fi

# File is executable
if [ -x "/bin/bash" ]; then
    echo "File is executable"
fi

# File has content
if [ -s "$file" ]; then
    echo "File has content"
fi
```

### **Getting File Information**

```bash
#!/bin/bash

file="/etc/hostname"

# File size
stat -f%z "$file"  # macOS
stat -c%s "$file"  # Linux

# File permissions
stat -f%Lp "$file"  # macOS
stat -c%a "$file"   # Linux

# File owner
stat -f%Su "$file"  # macOS
stat -c%U "$file"   # Linux

# Last modification time
stat -f%Sm "$file"  # macOS
stat -c%y "$file"   # Linux
```

## **Part 6: File Locking and Atomic Operations**

### **Ensuring Atomicity**

```bash
#!/bin/bash

# Atomic file write
write_atomic() {
    local file=$1
    local content=$2
    local temp=$(mktemp)
    
    # Write to temporary file
    echo "$content" > "$temp"
    
    # Atomically move
    mv "$temp" "$file"
}

write_atomic "output.txt" "Important data"
```

## **Troubleshooting Guide**

| Issue | Solution |
|-------|----------|
| File not found | Check path, use `[ -f "$file" ]` to verify |
| Permission denied | Check file permissions with `ls -l` |
| IFS not splitting | Ensure IFS is set before read command |
| Variable expansion in file | Use double quotes or process substitution |
| File locking issues | Use temp file and atomic move |

---

## **Best Practices**

1. **Quote file paths** - `"$file"` not `$file`
2. **Check file existence** before operations
3. **Use temp files** for atomic operations
4. **Set proper permissions** with chmod
5. **Backup before modification** - test first
6. **Use `-r` flag with read** to handle backslashes
7. **Close file descriptors** properly
8. **Use appropriate separators** with IFS
9. **Validate file content** before processing
10. **Clean up temp files** after operations
